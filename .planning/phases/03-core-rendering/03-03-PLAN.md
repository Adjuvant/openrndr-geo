    ```kotlin
    fun drawMultiPoint(drawer: Drawer, multiPoint: MultiPoint, userStyle: Style? = null) {

        }
    }
    ```

    DrawMultiLineString function:
    ```kotlin
    fun drawMultiLineString(drawer: Drawer, multiLineString: MultiLineString, userStyle: Style? = null) {
        multiLineString.lineStrings.forEach { lineString ->

        }
    }
    ```

    ```kotlin
    fun drawMultiPolygon(drawer: Drawer, multiPolygon: MultiPolygon, userStyle: Style? = null) {

        }
    }
    ```
