```kotlin
fun main() = application {
    configure {
        width = 1024
        height = 768
    }

    oliveProgram {
        // Load larger test dataset (GeoPackage has more features)
        val data = load("data/geo/ness-vectors.gpkg")
        val projection = ProjectionFactory.mercator(width = width, height = height)

        extend {
            drawer.clear(ColorRGBa.GRAY.shade(0.9))

            // Render all features with default styling
            data.features.forEach { feature ->
                val geometry = feature.geometry
                val screenGeom = when (geometry) {
                    is geo.Point -> {
                        Point(geometry.toScreen(projection).x, geometry.toScreen(projection).y)
                    }
                    is geo.LineString -> {
                        LineString(geometry.points.map { pt ->
                            Point(pt.x, pt.y).toScreen(projection)
                        })
                    }
                    is geo.Polygon -> {
                        Polygon(geometry.exterior.map { pt ->
                            Point(pt.x, pt.y).toScreen(projection)
                        })
                    }
                    else -> null
                }

                // Render with default styling
                when (screenGeom) {
                    is Point -> {
                        drawPoint(drawer, screenGeom.x, screenGeom.y, StyleDefaults.defaultPointStyle)
                    }
                    is LineString -> {
                        drawLineString(drawer, screenGeom.points, StyleDefaults.defaultLineStyle)
                    }
                    is Polygon -> {
                        drawPolygon(drawer, screenGeom.exterior, StyleDefaults.defaultPolygonStyle)
                    }
                }
            }
        }
    }
}
```

Important notes:
- Follow TemplateLiveProgram.kt: use oliveProgram {} instead of program {}
