    Create mergeStyles function:
    ```kotlin
    fun mergeStyles(default: Style, user: Style?): Style {
        if (user == null) return default.copy()  // Make copy to avoid mutating default
        return Style(
            fill = user.fill ?: default.fill,
            stroke = user.stroke ?: default.stroke,
            strokeWeight = user.strokeWeight.takeIf { it != 0.0 } ?: default.strokeWeight,
            size = user.size.takeIf { it != 0.0 } ?: default.size,
            shape = user.shape,
            lineCap = user.lineCap,
            lineJoin = user.lineJoin,
            miterLimit = user.miterLimit
        )
    }
    ```

    Note: Using default.copy() returns a copy, not the default object itself. This ensures defaults aren't mutated.

    Create public drawing functions:
    ```kotlin
    fun drawPoint(drawer: Drawer, screenX: Double, screenY: Double, userStyle: Style? = null) {
        val style = mergeStyles(StyleDefaults.defaultPointStyle, userStyle)
        writePoint(drawer, screenX, screenY, style)
    }

    fun drawLineString(drawer: Drawer, points: List<Vector2>, userStyle: Style? = null) {
        val style = mergeStyles(StyleDefaults.defaultLineStyle, userStyle)
        writeLineString(drawer, points, style)
    }

    fun drawPolygon(drawer: Drawer, points: List<Vector2>, userStyle: Style? = null) {
        val style = mergeStyles(StyleDefaults.defaultPolygonStyle, userStyle)
        writePolygon(drawer, points, style)
    }
    ```

    Reference RESEARCH.md "Style Merge with User Override Precedence" section.
    Follow CONTEXT.md "System defaults + per-geometry merge" decision.
