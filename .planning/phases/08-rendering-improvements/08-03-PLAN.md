@Test
fun testInteriorsToScreen() {
    // Create a polygon with a hole
    val poly = Polygon(
        exterior = listOf(
            Vector2(0.0, 0.0),
            Vector2(100.0, 0.0),
            Vector2(100.0, 100.0),
            Vector2(0.0, 100.0)
        ),
        interiors = listOf(
            listOf(
                Vector2(25.0, 25.0),
                Vector2(75.0, 25.0),
                Vector2(75.0, 75.0),
                Vector2(25.0, 75.0)
            )
        )
    )
    
    // Use identity-like projection for predictable results
    val projection = geo.projection.ProjectionFactory.equirectangular(
        width = 800.0,
        height = 600.0
    )
    
    // Project interior rings
    val projectedInteriors = poly.interiorsToScreen(projection)
    
    // Verify structure
    assertEquals(1, projectedInteriors.size)  // One interior ring
    assertEquals(4, projectedInteriors[0].size)  // Four points in the hole
    
    // Verify coordinates are transformed (not equal to input)
    // The projection transforms geo coords to screen coords
    val firstHolePoint = projectedInteriors[0][0]
    assertNotNull(firstHolePoint)
    assertNotEquals(25.0, firstHolePoint.x)  // Should be projected
}

@Test

@Test
fun testMultiPolygonHoleClamping() {
    // Create a polygon with a hole at extreme latitude
    val polyWithHoleAtPole = geo.Polygon(
        exterior = listOf(
            Vector2(0.0, 89.0),
            Vector2(10.0, 89.0),
            Vector2(10.0, 91.0),  // Beyond Mercator limit
            Vector2(0.0, 91.0)    // Beyond Mercator limit
        ),
        interiors = listOf(
            listOf(
                Vector2(2.0, 89.5),
                Vector2(8.0, 89.5),
                Vector2(8.0, 92.0),  // Beyond Mercator limit
                Vector2(2.0, 92.0)   // Beyond Mercator limit
            )
        )
    )
    
    // Clamp the polygon
    val clamped = polyWithHoleAtPole.clampToMercator()
    
    // Verify it's still a Polygon
    assertTrue(clamped is geo.Polygon)
    
    val clampedPoly = clamped as geo.Polygon
    
    // Verify exterior is clamped
    val maxExteriorY = clampedPoly.exterior.maxOf { it.y }
    assertTrue(maxExteriorY <= geo.MAX_MERCATOR_COORD, "Exterior Y should be clamped to MAX_MERCATOR_COORD")
    
    // Verify interior is clamped
    val maxInteriorY = clampedPoly.interiors.flatten().maxOf { it.y }
    assertTrue(maxInteriorY <= geo.MAX_MERCATOR_COORD, "Interior Y should be clamped to MAX_MERCATOR_COORD")
}
