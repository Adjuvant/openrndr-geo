```kotlin
package geo

import kotlinx.serialization.SerialName
import kotlinx.serialization.Serializable
import kotlinx.serialization.decodeFromString
import kotlinx.serialization.json.Json
import kotlinx.serialization.json.JsonElement
import kotlinx.serialization.json.jsonObject
import kotlinx.serialization.json.jsonPrimitive
import org.openrndr.math.Vector2
import java.io.File
import java.io.FileNotFoundException

object GeoJSON {
    private val json = Json {
        ignoreUnknownKeys = true
        isLenient = true
    }

    @Serializable
    private data class GeoJSONFeature(
        val type: String,
        val geometry: JsonElement,
        val properties: Map<String, Any?>? = null
    )

    @Serializable
    private data class GeoJSONFeatureCollection(
        val type: String,
        val features: List<GeoJSONFeature>
    )

    fun load(path: String): GeoJSONSource {
        val file = File(path)
        if (!file.exists()) {
            throw FileNotFoundException("GeoJSON file not found: $path")
        }

        val content = file.readText()
        val collection = json.decodeFromString<GeoJSONFeatureCollection>(content)

        if (collection.type != "FeatureCollection") {
            throw IllegalArgumentException("Expected FeatureCollection, got: ${collection.type}")
        }

        val features = collection.features.mapNotNull { geoJsonFeature ->
            try {
                val geometry = parseGeometry(geoJsonFeature.geometry)
                Feature(
                    geometry = geometry,
                    properties = geoJsonFeature.properties ?: emptyMap()
                )
            } catch (e: Exception) {
                println("Warning: Failed to parse feature: ${e.message}")
                null
            }
        }

        return GeoJSONSource(features.asSequence())
    }

    fun loadString(content: String): GeoJSONSource {
        val collection = json.decodeFromString<GeoJSONFeatureCollection>(content)

        if (collection.type != "FeatureCollection") {
            throw IllegalArgumentException("Expected FeatureCollection, got: ${collection.type}")
        }

        val features = collection.features.mapNotNull { geoJsonFeature ->
            try {
                val geometry = parseGeometry(geoJsonFeature.geometry)
                Feature(
                    geometry = geometry,
                    properties = geoJsonFeature.properties ?: emptyMap()
                )
            } catch (e: Exception) {
                println("Warning: Failed to parse feature: ${e.message}")
                null
            }
        }

        return GeoJSONSource(features.asSequence())
    }

    private fun parseGeometry(json: JsonElement): Geometry {
        val type = json.jsonObject["type"]?.jsonPrimitive?.content
            ?: throw IllegalArgumentException("Missing 'type' in geometry")

        val coords = json.jsonObject["coordinates"]
            ?: throw IllegalArgumentException("Missing 'coordinates' in geometry")

        return when (type) {
            "Point" -> parsePoint(coords)
            "LineString" -> parseLineString(coords)
            "Polygon" -> parsePolygon(coords)
            "MultiPoint" -> parseMultiPoint(coords)
            "MultiLineString" -> parseMultiLineString(coords)
            "MultiPolygon" -> parseMultiPolygon(coords)
            else -> throw IllegalArgumentException("Unknown geometry type: $type")
        }
    }

    private fun parsePoint(coords: JsonElement): Geometry {
        val coordArray = coords.jsonArray
        if (coordArray.size < 2) {
            throw IllegalArgumentException("Point must have at least 2 coordinates, got ${coordArray.size}")
        }
        val x = coordArray[0].jsonPrimitive.double
        val y = coordArray[1].jsonPrimitive.double
        return Point(x, y)
    }

    private fun parseLineString(coords: JsonElement): Geometry {
        val coordArray = coords.jsonArray
        if (coordArray.size < 2) {
            throw IllegalArgumentException("LineString must have at least 2 points")
        }
        val points = coordArray.mapNotNull { coord ->
            val arr = coord.jsonArray
            if (arr.size >= 2) {
                Vector2(arr[0].jsonPrimitive.double, arr[1].jsonPrimitive.double)
            } else null
        }
        return LineString(points)
    }

    private fun parsePolygon(coords: JsonElement): Geometry {
        val rings = coords.jsonArray
        if (rings.isEmpty()) {
            throw IllegalArgumentException("Polygon must have at least 1 ring")
        }
        val exterior = rings[0].jsonArray.mapNotNull { coord ->
            val arr = coord.jsonArray
            if (arr.size >= 2) {
                Vector2(arr[0].jsonPrimitive.double, arr[1].jsonPrimitive.double)
            } else null
        }
        val interiors = rings.drop(1).map { ring ->
            ring.jsonArray.mapNotNull { coord ->
                val arr = coord.jsonArray
                if (arr.size >= 2) {
                    Vector2(arr[0].jsonPrimitive.double, arr[1].jsonPrimitive.double)
                } else null
            }
        }
        return Polygon(exterior, interiors)
    }

    private fun parseMultiPoint(coords: JsonElement): Geometry {
        val coordArray = coords.jsonArray
        val points = coordArray.mapNotNull { coord ->
            val arr = coord.jsonArray
            if (arr.size >= 2) {
                Point(arr[0].jsonPrimitive.double, arr[1].jsonPrimitive.double)
            } else null
        }
        return MultiPoint(points)
    }

    private fun parseMultiLineString(coords: JsonElement): Geometry {
        val lines = coords.jsonArray
        val lineStrings = lines.mapNotNull { line ->
            val arr = line.jsonArray
            if (arr.size >= 2) {
                val points = arr.mapNotNull { coord ->
                    val c = coord.jsonArray
                    if (c.size >= 2) {
                        Vector2(c[0].jsonPrimitive.double, c[1].jsonPrimitive.double)
                    } else null
                }
                LineString(points)
            } else null
        }
        return MultiLineString(lineStrings)
    }

    private fun parseMultiPolygon(coords: JsonElement): Geometry {
        val polygons = coords.jsonArray
        val polygonList = polygons.mapNotNull { poly ->
            val rings = poly.jsonArray
            if (rings.isNotEmpty()) {
                val exterior = rings[0].jsonArray.mapNotNull { coord ->
                    val arr = coord.jsonArray
                    if (arr.size >= 2) {
                        Vector2(arr[0].jsonPrimitive.double, arr[1].jsonPrimitive.double)
                    } else null
                }
                val interiors = rings.drop(1).map { ring ->
                    ring.jsonArray.mapNotNull { coord ->
                        val arr = coord.jsonArray
                        if (arr.size >= 2) {
                            Vector2(arr[0].jsonPrimitive.double, arr[1].jsonPrimitive.double)
                        } else null
                    }
                }
                Polygon(exterior, interiors)
            } else null
        }
        return MultiPolygon(polygonList)
    }
}

class GeoJSONSource(
    override val features: Sequence<Feature>,
    crs: String = "EPSG:4326"
) : GeoSource(crs) {
    companion object {
        @JvmStatic
        fun load(path: String): GeoJSONSource = GeoJSON.load(path)
    }
}
```
