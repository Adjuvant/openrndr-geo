```kotlin
fun main() = application {
    configure {
        width = 800
        height = 600
    }

    program {
        // Load geo data
        val data = load("data/sample.geojson")
        val projection = ProjectionFactory.mercator(width = width, height = height)

        extend {
            drawer.clear(ColorRGBa.WHITE)

            // Render features
            data.features.forEach { feature ->
                val geometry = feature.geometry

                // Project and render based on geometry type
                when (geometry) {
                    is geo.Point -> {
                        val screen = geometry.toScreen(projection)
                        drawPoint(drawer, screen.x, screen.y, Style {
                            size = 10.0
                            shape = Shape.Circle
                            fill = ColorRGBa.RED
                        })
                    }
                    is geo.LineString -> {
                        val screenPoints = geometry.points.map { pt ->
                            // Note: LineString points are already Vector2 in local coords
                            // Create Point to use toScreen()
                            geo.Point(pt.x, pt.y).toScreen(projection)
                        }
                        drawLineString(drawer, screenPoints, Style {
                            stroke = ColorRGBa.BLUE
                            strokeWeight = 2.0
                        })
                    }
                    // Handle other geometry types as needed
                }
            }
        }
    }
}
```

Important notes:
- data/sample.geojson contains Point and LineString features
- LineString points are Vector2, need conversion to Point for toScreen()
- Follow TemplateProgram.kt: configure {} block then program {} block
- Add comments explaining each part (loading, projection, rendering)
- Keep it simple and clear for learning purposes
